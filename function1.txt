const rawDate = msg.payload.order_date;

// ---- Clean date ----
const cleanedDate = rawDate
  .replace(/^[A-Za-z]+,\s*/, '')
  .replace(/\sIST$/, '');

const parsedDate = new Date(cleanedDate + ' GMT+0530');
const eventTime = parsedDate.getTime(); // ✅ epoch millis


// ---- Clean city & state ----
const cleanCity = msg.payload.ship_city
  ? msg.payload.ship_city.replace(/,+$/, '').trim().toUpperCase()
  : null;

const cleanState = msg.payload.ship_state
  ? msg.payload.ship_state.replace(/,+$/, '').trim().toUpperCase()
  : null;

// ---- Clean amount (₹, commas) ----
const cleanAmount = msg.payload.item_total
  ? Number(msg.payload.item_total.replace(/[₹,]/g, ''))
  : null;

const order = {
  order_id: msg.payload.order_no,
  order_date: rawDate,
  buyer_name: msg.payload.buyer,
  city: cleanCity,          // ✅ FIXED
  state: cleanState,        // ✅ FIXED
  sku: msg.payload.sku,
  description: msg.payload.description,
  quantity: Number(msg.payload.quantity),
  amount: cleanAmount,      // ✅ FIXED
  event_time: eventTime
};

msg.payload = JSON.stringify(order);
return msg;
